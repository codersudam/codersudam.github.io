  <script>
    const canvas = document.getElementById('liquidCanvas');
    const ctx = canvas.getContext('2d');

    let width, height;
    let springs = [];
    const spacing = 8; // Slightly wider spacing for smoother, larger waves

    // Upgraded Spring Physics Constants for a "heavier" water feel
    const tension = 0.050;
    const dampening = 0.010;
    const spread = 0.3;

    let targetHeight;
    let mouse = { x: 0, y: 0, dy: 0, prevY: 0 };

    class Spring {
      constructor(x, y) {
        this.x = x;
        this.targetHeight = y;
        this.height = y;
        this.velocity = 0;
      }

      update() {
        let displacement = this.targetHeight - this.height;
        this.velocity += tension * displacement - this.velocity * dampening;
        this.height += this.velocity;
      }
    }

    function initFluid() {
      width = canvas.width = window.innerWidth;
      height = canvas.height = window.innerHeight;
      
      // Set water level lower for the light theme aesthetic
      targetHeight = height * 0.65; 
      
      springs = [];
      const numSprings = Math.ceil(width / spacing) + 1;
      
      for (let i = 0; i < numSprings; i++) {
        springs.push(new Spring(i * spacing, targetHeight));
      }
    }

    function drawFluid() {
      ctx.clearRect(0, 0, width, height);

      // --- Interaction Logic ---
      let index = Math.floor(mouse.x / spacing);
      if (index >= 0 && index < springs.length) {
        let distance = Math.abs(mouse.y - springs[index].height);
        // Push water if mouse crosses the surface
        if (distance < 60) {
          springs[index].velocity += mouse.dy * 1;
        }
      }
      mouse.dy *= 0.8; // Decay velocity

      // --- Physics Update ---
      for (let i = 0; i < springs.length; i++) {
        springs[i].update();
      }

      // Propagate the wave
      let leftDeltas = new Array(springs.length).fill(0);
      let rightDeltas = new Array(springs.length).fill(0);

      for (let j = 0; j < 8; j++) {
        for (let i = 0; i < springs.length; i++) {
          if (i > 0) {
            leftDeltas[i] = spread * (springs[i].height - springs[i - 1].height);
            springs[i - 1].velocity += leftDeltas[i];
          }
          if (i < springs.length - 1) {
            rightDeltas[i] = spread * (springs[i].height - springs[i + 1].height);
            springs[i + 1].velocity += rightDeltas[i];
          }
        }
        for (let i = 0; i < springs.length; i++) {
          if (i > 0) springs[i - 1].height += leftDeltas[i];
          if (i < springs.length - 1) springs[i + 1].height += rightDeltas[i];
        }
      }

      // --- Rendering the Light Theme Liquid ---
      
      // Beautiful crystal blue gradient
      let gradient = ctx.createLinearGradient(0, targetHeight - 100, 0, height);
      gradient.addColorStop(0, 'rgba(56, 189, 248, 0.4)'); // Light sky blue
      gradient.addColorStop(1, 'rgba(14, 165, 233, 0.8)'); // Deeper cerulean

      ctx.beginPath();
      ctx.moveTo(0, height);
      ctx.lineTo(springs[0].x, springs[0].height);

      for (let i = 0; i < springs.length - 1; i++) {
        let p1 = springs[i];
        let p2 = springs[i + 1];
        let cx = (p1.x + p2.x) / 2;
        let cy = (p1.height + p2.height) / 2;
        ctx.quadraticCurveTo(p1.x, p1.height, cx, cy);
      }
      
      ctx.lineTo(width, springs[springs.length - 1].height);
      ctx.lineTo(width, height);
      ctx.closePath();
      
      ctx.fillStyle = gradient;
      ctx.fill();

      // Top edge crisp highlight
      ctx.beginPath();
      ctx.moveTo(springs[0].x, springs[0].height);
      for (let i = 0; i < springs.length - 1; i++) {
        let p1 = springs[i];
        let p2 = springs[i + 1];
        let cx = (p1.x + p2.x) / 2;
        let cy = (p1.height + p2.height) / 2;
        ctx.quadraticCurveTo(p1.x, p1.height, cx, cy);
      }
      ctx.lineWidth = 4;
      ctx.strokeStyle = 'rgba(255, 255, 255, 0.8)';
      ctx.stroke();

      requestAnimationFrame(drawFluid);
    }

    // --- Event Listeners ---
    window.addEventListener('resize', initFluid);

    window.addEventListener('mousemove', (e) => {
      mouse.x = e.clientX;
      mouse.y = e.clientY;
      mouse.dy = mouse.y - mouse.prevY;
      mouse.prevY = mouse.y;
    });

    window.addEventListener('touchmove', (e) => {
      mouse.x = e.touches[0].clientX;
      mouse.y = e.touches[0].clientY;
      mouse.dy = mouse.y - mouse.prevY;
      mouse.prevY = mouse.y;
    });

    // Idle splashes
    setInterval(() => {
        if(Math.abs(mouse.dy) < 1) {
            let randomIndex = Math.floor(Math.random() * springs.length);
            springs[randomIndex].velocity -= Math.random() * 25 + 15;
        }
    }, 2500);

    initFluid();
    drawFluid();
  </script>
